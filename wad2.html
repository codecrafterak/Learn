HTML5 · Storage

Web Storage API

The Web Storage API provides two mechanisms — localStorage and sessionStorage — to store key-value string data in the browser. Unlike cookies, storage data is not sent with every request and has much larger limits (typically 5–10 MB per origin).

JavaScript — Web Storage
// ─── localStorage ───
// Persists FOREVER until explicitly cleared

// Store a string
localStorage.setItem('username', 'alice');

// Store an object (must stringify)
const user = { name: 'Alice', age: 28 };
localStorage.setItem('user', JSON.stringify(user));

// Read
const raw = localStorage.getItem('user');
const parsed = JSON.parse(raw); // { name:'Alice', age:28 }

// Remove one key
localStorage.removeItem('username');

// Clear all localStorage
localStorage.clear();

// Number of items
console.log(localStorage.length);

// Iterate all keys
for (let i = 0; i < localStorage.length; i++) {
  const key = localStorage.key(i);
  console.log(key, localStorage.getItem(key));
}

// ─── sessionStorage ───
// Cleared when the TAB is closed
sessionStorage.setItem('token', 'abc123');
sessionStorage.getItem('token');
sessionStorage.removeItem('token');

// ─── storage EVENT ───
// Fires in OTHER tabs when localStorage changes
window.addEventListener('storage', (e) => {
  console.log(e.key);      // changed key
  console.log(e.oldValue); // previous value
  console.log(e.newValue); // new value
  console.log(e.url);      // url that changed it
});

Geolocation API

The Geolocation API lets web pages request a user's geographic location. It always requires explicit user permission. Accuracy depends on available methods: GPS, Wi-Fi triangulation, cell towers, or IP address.

JavaScript — Geolocation API
// ─── Check support ───
if ('geolocation' in navigator) {
  console.log('Geolocation supported!');
}

// ─── Get current position (one-time) ───
navigator.geolocation.getCurrentPosition(
  // Success callback
  (position) => {
    const { latitude, longitude, accuracy } = position.coords;
    console.log(`Lat: ${latitude}, Lng: ${longitude}`);
    console.log(`Accuracy: ±${accuracy}m`);
    // Other available coords:
    // altitude, altitudeAccuracy, heading, speed
    console.log(position.timestamp); // Unix ms
  },
  // Error callback
  (error) => {
    switch(error.code) {
      case 1: console.log('Permission denied'); break;
      case 2: console.log('Position unavailable'); break;
      case 3: console.log('Timeout'); break;
    }
  },
  // Options
  {
    enableHighAccuracy: true,  // Use GPS if available
    timeout: 5000,             // ms before error
    maximumAge: 60000,         // Cache location 60s
  }
);

// ─── Watch position (continuous updates) ───
const watchId = navigator.geolocation.watchPosition(
  (position) => {
    console.log('Updated: ', position.coords.latitude);
  }
);

// Stop watching
navigator.geolocation.clearWatch(watchId);
⚠️ Permissions Required

Geolocation requires explicit user consent and only works on HTTPS origins (or localhost for development). Mobile devices typically use GPS; desktops use Wi-Fi/IP which is far less accurate.

Web Workers

Web Workers run JavaScript in a background thread, completely separate from the main UI thread. This prevents heavy computations from blocking the page and freezing the UI. Workers communicate with the main thread via message passing.

JavaScript — Web Workers
// ─── Main thread (index.js) ───

// Create a worker from a JS file
const worker = new Worker('worker.js');

// Send data TO the worker
worker.postMessage({ task: 'compute', data: [1,2,3] });

// Receive data FROM the worker
worker.addEventListener('message', (event) => {
  console.log('Worker result:', event.data);
});

// Handle errors
worker.addEventListener('error', (err) => {
  console.error('Worker error:', err.message);
});

// Terminate the worker
worker.terminate();

// ─────────────────────────────────────
// worker.js (runs in background thread)
// ─────────────────────────────────────

// Receive messages from main thread
self.addEventListener('message', (event) => {
  const { task, data } = event.data;

  if (task === 'compute') {
    // Heavy computation — won't freeze UI!
    const result = data.reduce((a, b) => a + b, 0);

    // Send result back to main thread
    self.postMessage({ result });
  }
});

// ─── Inline Worker (no separate file) ───
const blob = new Blob([
  `self.onmessage = e => self.postMessage(e.data * 2)`
], { type: 'text/javascript' });

const inlineWorker = new Worker(URL.createObjectURL(blob));
inlineWorker.postMessage(21);  // → receives 42

// ─── SharedWorker (shared across tabs) ───
const shared = new SharedWorker('shared.js');
shared.port.postMessage('hello');
shared.port.onmessage = (e) => console.log(e.data);

WebSockets

WebSockets provide a persistent, full-duplex (two-way) communication channel between the browser and a server over a single TCP connection. Unlike HTTP's request/response model, either party can send data at any time — ideal for real-time apps like chat, live dashboards, and multiplayer games.

JavaScript — WebSocket API
// ─── Create connection ───
const ws = new WebSocket('wss://example.com/socket');
// wss:// = secure (like https), ws:// = insecure

// ─── Connection opened ───
ws.addEventListener('open', () => {
  console.log('Connected!');
  ws.send('Hello server');
  ws.send(JSON.stringify({ type: 'join', room: '123' }));
});

// ─── Receive messages ───
ws.addEventListener('message', (event) => {
  console.log('Received:', event.data);
  const msg = JSON.parse(event.data);
});

// ─── Connection closed ───
ws.addEventListener('close', (event) => {
  console.log('Closed. Code:', event.code);
  // Reconnect logic here
});

// ─── Handle errors ───
ws.addEventListener('error', (err) => {
  console.error('WebSocket error', err);
});

// ─── Send data ───
ws.send('plain text');
ws.send(JSON.stringify({ message: 'hi' }));
ws.send(new ArrayBuffer(8));  // Binary data

// ─── Connection state ───
// ws.readyState: 0=CONNECTING, 1=OPEN, 2=CLOSING, 3=CLOSED
if (ws.readyState === WebSocket.OPEN) ws.send('safe to send');

// ─── Close connection ───
ws.close();
ws.close(1000, 'Normal closure');

History API

The History API lets JavaScript manipulate the browser's session history (the pages visited in the current tab). It enables Single Page Applications (SPAs) to change the URL without triggering a full page reload, creating seamless navigation with shareable, bookmarkable URLs.

JavaScript — History API
// ─── pushState: add new entry to history ───
// Does NOT trigger page reload!
history.pushState(
  { page: 2, item: 'shoes' },  // state object
  '',                          // title (ignored by browsers)
  '/products/shoes'            // new URL
);
// URL bar now shows /products/shoes

// ─── replaceState: replace current entry ───
history.replaceState(
  { page: 1 },
  '',
  '/products/shoes?color=red'
);

// ─── Navigate programmatically ───
history.back();       // go back one page
history.forward();    // go forward one page
history.go(-2);       // go back 2 pages
history.go(1);        // go forward 1 page

// ─── History length ───
console.log(history.length);  // number of entries

// ─── popstate: fires on back/forward button ───
window.addEventListener('popstate', (event) => {
  console.log('Navigated to:', location.pathname);
  console.log('State:', event.state); // restored state obj
  // Re-render based on event.state or URL
});

// ─── SPA Router pattern ───
function navigate(path, state = {}) {
  history.pushState(state, '', path);
  renderPage(path);  // your render function
}
navigate('/about');

Custom Data Attributes

HTML5's data-* attributes allow you to store private custom data on any HTML element. They're perfect for embedding data in markup that JavaScript needs to read — without using hidden inputs, classes, or non-standard attributes.

HTML + JavaScript — Data Attributes
<!-- HTML: data-[name]="value" ─── -->
<div
  id="product"
  data-product-id="42"
  data-category="electronics"
  data-price="299.99"
  data-in-stock="true"
>MacBook Pro</div>

<button data-action="add-to-cart"
        data-product-id="42">Add</button>

// JavaScript: read via .dataset ───
const el = document.getElementById('product');

// dataset converts kebab-case to camelCase
el.dataset.productId   // "42"  (note: always strings!)
el.dataset.category    // "electronics"
el.dataset.price       // "299.99"
el.dataset.inStock     // "true"

// Parse to appropriate types
const id = parseInt(el.dataset.productId, 10);
const price = parseFloat(el.dataset.price);
const inStock = el.dataset.inStock === 'true';

// Set/update
el.dataset.price = '249.99';

// Delete
delete el.dataset.category;

// CSS attribute selector (very useful!)
/* [data-category="electronics"] { background: blue; } */
/* [data-in-stock="false"] { opacity: 0.5; } */

// Event delegation with data attributes
document.addEventListener('click', (e) => {
  const action = e.target.dataset.action;
  if (action === 'add-to-cart') {
    addToCart(e.target.dataset.productId);
  }
});

ARIA & Accessibility

ARIA (Accessible Rich Internet Applications) is a set of attributes that extend HTML's semantics for assistive technologies. Combined with semantic HTML, ARIA makes interactive widgets accessible to users of screen readers, keyboard-only navigation, and other assistive tech.

ARIA Roles, States & Properties WAI-ARIA

HTML — ARIA Attributes
<!-- ─── ROLES: what the element IS ─── -->
<div role="button" tabindex="0">Click me</div>
<div role="navigation">...</div>
<div role="dialog" aria-modal="true">...</div>
<div role="alert">Error: Form invalid</div>
<div role="progressbar"
     aria-valuenow="65"
     aria-valuemin="0"
     aria-valuemax="100"></div>
<ul role="listbox">
  <li role="option" aria-selected="true">Item A</li>
</ul>

<!-- ─── LABELS & DESCRIPTIONS ─── -->
<!-- aria-label: direct string label -->
<button aria-label="Close dialog">✕</button>

<!-- aria-labelledby: reference another element -->
<h2 id="dialog-title">Settings</h2>
<div role="dialog" aria-labelledby="dialog-title">...</div>

<!-- aria-describedby: additional description -->
<input type="email" aria-describedby="email-hint">
<p id="email-hint">Format: you@example.com</p>

<!-- ─── STATES ─── -->
<button aria-pressed="true">Dark Mode</button>
<button aria-expanded="false" aria-controls="menu">Menu</button>
<div aria-hidden="true">Decorative (hidden from SR)</div>
<span aria-live="polite">Loading results...</span>
<span aria-live="assertive">ERROR: Payment failed</span>
<input aria-required="true" aria-invalid="true">

<!-- ─── FOCUS MANAGEMENT ─── -->
<!-- tabindex: make non-focusable elements focusable -->
<div tabindex="0">Focusable div</div>
<div tabindex="-1">Programmatically focusable only</div>
<!-- Positive tabindex (avoid): creates tab order -->
<input tabindex="1"> <!-- First in tab order -->

<!-- Skip navigation link -->
<a href="#main-content" class="skip-link">Skip to main</a>

Always prefer native semantic HTML over ARIA. Use <button> not <div role="button">. ARIA should fill gaps where no semantic HTML element exists, not replace it.

Dialog & Popover

HTML5's native <dialog> element creates accessible modal and non-modal dialogs with built-in backdrop, focus trapping, and Escape-key closing — no JavaScript libraries needed. The newer popover API provides lightweight, non-modal overlays.

HTML + JavaScript — Dialog
<!-- Modal dialog -->
<dialog id="myDialog">
  <h3>Confirm Action</h3>
  <p>Are you sure?</p>
  <!-- method="dialog" closes dialog on submit -->
  <form method="dialog">
    <button value="cancel">Cancel</button>
    <button value="confirm">Confirm</button>
  </form>
</dialog>

// JavaScript control
const dialog = document.getElementById('myDialog');

// .show() = non-modal (no backdrop)
dialog.show();

// .showModal() = modal (with backdrop)
dialog.showModal();

// Close programmatically
dialog.close();
dialog.close('my-return-value');

// Get return value
dialog.addEventListener('close', () => {
  console.log(dialog.returnValue); // "confirm" or "cancel"
});

<!-- Popover API (no JS needed for basic use!) -->
<button popovertarget="tip">Show tooltip</button>
<div id="tip" popover>
  I'm a popover! Press Esc or click outside to close.
</div>

<!-- popover types -->
<div popover="auto">Light dismiss</div>
<div popover="manual">Manual only</div>

Template & Slot Elements

The <template> element holds HTML that is not rendered on page load but can be cloned and inserted via JavaScript. <slot> elements are used inside Web Components to define insertion points for content provided by the component's user.

HTML + JavaScript — Template
<!-- template: not rendered until cloned ─── -->
<template id="card-template">
  <div class="card">
    <h3 class="card-title">Title</h3>
    <p class="card-body">Body text</p>
    <button class="card-btn">Learn More</button>
  </div>
</template>

// Clone and populate ─── 
const tmpl = document.getElementById('card-template');

function createCard(title, body) {
  // Clone the template content (deep clone)
  const clone = tmpl.content.cloneNode(true);

  // Populate content
  clone.querySelector('.card-title').textContent = title;
  clone.querySelector('.card-body').textContent = body;

  // Insert into the DOM
  document.body.appendChild(clone);
}

createCard('HTML5 Guide', 'Learn modern HTML5 APIs.');

<!-- slot: placeholder in a web component ─── -->
<template id="user-card-tmpl">
  <style>
    .wrapper { border: 1px solid #ccc; padding: 16px; }
  </style>
  <div class="wrapper">
    <!-- Default slot -->
    <slot>Default content if nothing provided</slot>
    <!-- Named slot -->
    <slot name="avatar">🧑 Default Avatar</slot>
  </div>
</template>

<!-- Consumer uses slot -->
<user-card>
  <img slot="avatar" src="avatar.jpg">
  <p>Alice — Developer</p>
</user-card>

Web Components

Web Components are a suite of browser APIs that let you create reusable, encapsulated custom HTML elements with their own styles and behavior. They consist of three technologies: Custom Elements, Shadow DOM, and HTML Templates.

JavaScript — Custom Elements & Shadow DOM
// ─── Define a custom element ───
class MyToast extends HTMLElement {
  // Which attributes to observe
  static observedAttributes = ['message', 'type'];

  constructor() {
    super();
    // Create shadow DOM: 'open' = accessible externally
    this.shadow = this.attachShadow({ mode: 'open' });
  }

  // ─── Lifecycle callbacks ───
  connectedCallback() {
    // Called when element is added to DOM
    this.render();
  }

  disconnectedCallback() {
    // Called when element is removed
    this.cleanup();
  }

  attributeChangedCallback(name, oldValue, newValue) {
    // Called when observed attribute changes
    this.render();
  }

  render() {
    const msg = this.getAttribute('message') || 'Hello!';
    const type = this.getAttribute('type') || 'info';

    // Styles are SCOPED to shadow DOM
    this.shadow.innerHTML = `
      <style>
        :host { display: block; padding: 12px; }
        .toast { border-radius: 6px; padding: 12px 16px; }
        .toast.info { background: #e3f2fd; color: #1565c0; }
        .toast.error { background: #ffebee; color: #c62828; }
      </style>
      <div class="toast ${type}">${msg}</div>
    `;
  }
}

// Register the custom element (must contain a hyphen!)
customElements.define('my-toast', MyToast);

// ─── Usage in HTML ───
// <my-toast message="Saved!" type="info"></my-toast>

// ─── Creating programmatically ───
const toast = document.createElement('my-toast');
toast.setAttribute('message', 'Upload complete!');
document.body.appendChild(toast);

// ─── Extending built-in elements ───
class FancyButton extends HTMLButtonElement {
  connectedCallback() {
    this.style.borderRadius = '999px';
  }
}
customElements.define('fancy-button', FancyButton, { extends: 'button' });

Intersection Observer

The Intersection Observer API asynchronously watches whether elements enter or leave the viewport (or another element). It replaces scroll-event hacks with a performant, declarative approach — ideal for lazy loading images, triggering animations on scroll, and infinite scroll.

JavaScript — Intersection Observer
// ─── Basic usage ───
const observer = new IntersectionObserver(
  // Callback: array of observed entries
  (entries) => {
    entries.forEach((entry) => {
      if (entry.isIntersecting) {
        // Element entered the viewport
        entry.target.classList.add('visible');
        observer.unobserve(entry.target); // stop watching
      }
    });
  },
  // Options
  {
    root: null,          // null = viewport
    rootMargin: '0px',   // expand/shrink root box
    threshold: 0.1,       // 0–1: % of element visible
    // threshold: [0, 0.5, 1] = fire at 0%, 50%, 100%
  }
);

// Observe elements
document.querySelectorAll('.animate-on-scroll')
  .forEach((el) => observer.observe(el));

// Stop observing
observer.unobserve(el);
observer.disconnect(); // stop all

// ─── Lazy load images ───
const lazyObserver = new IntersectionObserver((entries) => {
  entries.forEach((entry) => {
    if (entry.isIntersecting) {
      const img = entry.target;
      img.src = img.dataset.src;  // set real src
      lazyObserver.unobserve(img);
    }
  });
}, { rootMargin: '200px' }); // preload 200px early

document.querySelectorAll('img[data-src]')
  .forEach((img) => lazyObserver.observe(img));

// ─── CSS: animate elements on scroll ───
/* .animate-on-scroll { opacity: 0; transform: translateY(30px); transition: 0.6s ease; }
   .animate-on-scroll.visible { opacity: 1; transform: translateY(0); } */

Mutation Observer & Resize Observer

The Mutation Observer watches for changes to the DOM tree — added/removed nodes, attribute changes, text changes. The Resize Observer watches when an element's size changes, which is more reliable than window resize events for responsive components.

JavaScript — Mutation & Resize Observers
// ─── MutationObserver ───
const mutObs = new MutationObserver((mutations) => {
  mutations.forEach((mutation) => {
    console.log('Type:', mutation.type);
    // 'childList' | 'attributes' | 'characterData'

    if (mutation.type === 'childList') {
      mutation.addedNodes.forEach((node) => {
        console.log('Added:', node);
      });
      mutation.removedNodes.forEach((node) => {
        console.log('Removed:', node);
      });
    }

    if (mutation.type === 'attributes') {
      console.log('Changed attr:', mutation.attributeName);
      console.log('Old value:', mutation.oldValue);
    }
  });
});

mutObs.observe(document.body, {
  childList: true,   // watch added/removed children
  subtree: true,     // watch all descendants too
  attributes: true,  // watch attribute changes
  attributeOldValue: true,
  characterData: true,  // watch text changes
});

mutObs.disconnect(); // stop watching

// ─── ResizeObserver ───
const resObs = new ResizeObserver((entries) => {
  entries.forEach((entry) => {
    const { width, height } = entry.contentRect;
    console.log(`Element: ${width}×${height}px`);

    // Conditional rendering based on component width
    if (width < 400) {
      entry.target.classList.add('compact');
    } else {
      entry.target.classList.remove('compact');
    }
  });
});

resObs.observe(document.querySelector('.chart-container'));
resObs.unobserve(el);
resObs.disconnect();

Fetch API & XMLHttpRequest

The Fetch API is the modern way to make HTTP requests from the browser using Promises. It replaced the older XMLHttpRequest (XHR) with a cleaner, more powerful interface.

JavaScript — Fetch API
// ─── Basic GET request ───
const response = await fetch('https://api.example.com/users');
const data = await response.json();
console.log(data);

// ─── POST with JSON body ───
const res = await fetch('/api/users', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer token123',
  },
  body: JSON.stringify({ name: 'Alice', age: 28 }),
});

// ─── Check status ───
if (!res.ok) {  // ok = true if status 200-299
  throw new Error(`HTTP error: ${res.status}`);
}
const created = await res.json();

// ─── Response methods ───
response.json();       // Parse as JSON
response.text();       // Parse as text
response.blob();       // Parse as Blob (file)
response.arrayBuffer();// Parse as ArrayBuffer
response.formData();   // Parse as FormData
response.headers.get('Content-Type');

// ─── Abort / timeout ───
const controller = new AbortController();
setTimeout(() => controller.abort(), 5000);
const r = await fetch('/api', { signal: controller.signal });

// ─── Upload file ───
const form = new FormData();
form.append('file', fileInput.files[0]);
await fetch('/upload', { method: 'POST', body: form });

// ─── Parallel requests ───
const [users, posts] = await Promise.all([
  fetch('/api/users').then(r => r.json()),
  fetch('/api/posts').then(r => r.json()),
]);

CSS3 · Layout

CSS Positioning

The position property controls how an element is placed in the document flow and how it responds to top/right/bottom/left offsets. Understanding positioning is fundamental to creating overlays, fixed headers, tooltips, and complex layouts.

CSS — Position Values
/* ─── static (default) ─── */
.normal {
  position: static;
  /* top/right/bottom/left have NO effect */
}

/* ─── relative ─── */
.relative {
  position: relative;
  /* Offsets from its NORMAL position */
  top: 20px;   /* moves DOWN 20px */
  left: 10px;  /* moves RIGHT 10px */
  /* Still occupies original space in flow */
  /* Creates new stacking context */
  /* Acts as reference for absolute children */
}

/* ─── absolute ─── */
.absolute {
  position: absolute;
  /* Removed from document flow */
  /* Positioned relative to nearest POSITIONED ancestor */
  /* (relative/absolute/fixed/sticky) */
  /* Falls back to <html> if none found */
  top: 0; right: 0;      /* top-right corner */
  top: 50%; left: 50%;
  transform: translate(-50%, -50%); /* perfect center */
}

/* ─── fixed ─── */
.fixed {
  position: fixed;
  /* Positioned relative to the VIEWPORT */
  /* Stays in place during scroll */
  top: 0; left: 0; right: 0;  /* sticky header */
  width: 100%;
}

/* ─── sticky ─── */
.sticky {
  position: sticky;
  /* Hybrid: relative until scroll threshold */
  /* Then acts like fixed within its parent */
  top: 0;  /* sticks when within 0px of top */
  /* PARENT must have scrollable overflow */
  /* Parent must not have overflow:hidden */
}

Z-Index & Stacking Contexts

The z-index property controls the stack order (depth) of positioned elements. A stacking context is a group of elements that share the same z-index space. Understanding when new stacking contexts are created is key to debugging z-index issues.

CSS — Z-Index
/* z-index only works on POSITIONED elements */
/* (relative, absolute, fixed, sticky) */

.backdrop { position: fixed; z-index: 100; }
.modal    { position: fixed; z-index: 200; }
.tooltip  { position: absolute; z-index: 300; }

/* Stacking context is created by: */
.creates-context {
  /* 1. position + z-index ≠ auto */

The display property is the most fundamental CSS layout property — it defines the display type of an element, controlling how it and its children participate in layout. visibility and opacity control whether an element is seen without removing it from layout.

Overflow

The overflow property controls what happens when content exceeds its container's dimensions. It also creates a new block formatting context (BFC) and establishes a scrolling context.

The clip-path property clips an element to a defined shape — only the area inside the shape is visible. It can create polygons, circles, ellipses, and even follow SVG paths. shape-outside makes inline content flow around custom shapes.

CSS Counters let you number elements automatically using CSS alone. The content property (used in ::before and ::after) inserts generated text, symbols, counters, or attributes into the page without modifying the HTML.

CSS Math Functions

CSS math functions allow responsive, constraint-based sizing directly in CSS — no JavaScript or media queries needed for many common responsive patterns. clamp(), calc(), min(), and max() are powerful tools for fluid typography and adaptive layouts.

CSS — calc, clamp, min, max
/* ─── calc() ─── */
/* Mix units! Evaluated at runtime. */
.sidebar {
  width: calc(100% - 280px);        /* fill remaining space */
  height: calc(100vh - 60px);       /* below 60px header */
  padding: calc(2rem + 5px);
  font-size: calc(1rem * 1.25);
  margin-left: calc(-1 * var(--spacing));
}

/* ─── clamp(MIN, IDEAL, MAX) ─── */
/* Most powerful for fluid typography! */
h1 {
  /* min 24px, ideal 5vw, max 60px */
  font-size: clamp(24px, 5vw, 60px);
}

.container {
  width: clamp(320px, 90%, 1200px);
  /* never narrower than 320px, never wider than 1200px */
}

p {
  /* Fluid line height based on font size */
  line-height: clamp(1.3, calc(1.3 + 0.5vw), 1.8);
}

/* ─── min(A, B) ─── */
/* Returns the SMALLEST value */
.responsive {
  width: min(500px, 100%); /* = max-width: 500px */
  font-size: min(3vw, 18px);
}

/* ─── max(A, B) ─── */
/* Returns the LARGEST value */
.minimum {
  width: max(300px, 50%); /* = min-width: 300px */
  padding: max(12px, 2vw);
}

/* ─── Combining functions ─── */
.advanced {
  font-size: clamp(
    1rem,
    calc(1rem + 1vw),  /* fluid ideal */
    2rem
  );
  gap: clamp(1rem, 3%, 2rem);
}

CSS3 · At-Rules

@supports, @layer, @scope

Modern CSS at-rules give you powerful tools for feature detection, managing style priority, and scoping styles to specific DOM subtrees — capabilities that previously required preprocessors or JavaScript.

CSS — @supports, @layer, @scope
/* ─── @supports: feature detection ─── */
@supports (display: grid) {
  /* Only applies if grid is supported */
  .layout { display: grid; }
}

@supports not (display: grid) {
  /* Fallback for older browsers */
  .layout { display: flex; }
}

@supports (backdrop-filter: blur(1px)) {
  .glass { backdrop-filter: blur(10px); }
}

@supports (selector(:has(a))) {
  /* :has() pseudo-class is supported */
  div:has(> img) { border: 2px solid blue; }
}

@supports (color: oklch(0 0 0)) {
  :root { --primary: oklch(0.7 0.2 60); }
}

/* ─── @layer: cascade layers ─── */
/* Declare layer order (earlier = lower priority) */
@layer reset, base, components, utilities;

/* Styles in 'reset' layer — lowest priority */
@layer reset {
  * { margin: 0; padding: 0; }
}

/* Styles in 'base' layer */
@layer base {
  body { font-family: Georgia, serif; }
}

/* 'utilities' layer overrides 'components' REGARDLESS of specificity */
@layer utilities {
  .mt-4 { margin-top: 1rem !important; }
}

/* Unlayered styles have highest priority */
.override { color: red; } /* beats all layers */

/* ─── @scope (CSS 2023+) ─── */
@scope (.card) {
  /* Styles only apply inside .card */
  h2 { color: navy; }   /* won't affect other h2s */
  p  { color: gray; }
}

/* @scope with lower boundary */
@scope (.card) to (.card-footer) {
  /* Only elements between .card and .card-footer */
  a { color: blue; }
}

CSS Logical Properties

Logical properties replace physical directional properties (left/right/top/bottom) with writing-mode-relative equivalents (inline/block, start/end). They automatically adapt to RTL (right-to-left) languages and vertical writing modes — essential for truly international, multilingual layouts.

CSS — Logical Properties
/* Physical → Logical equivalents */

/* ─── Margin ─── */
/* margin-top     → */  margin-block-start: 1rem;
/* margin-bottom  → */  margin-block-end: 1rem;
/* margin-left    → */  margin-inline-start: auto;
/* margin-right   → */  margin-inline-end: auto;
/* margin: 0 auto → */  margin-inline: auto;
                         margin-block: 0;

/* ─── Padding ─── */
padding-block: 16px;          /* top + bottom */
padding-inline: 24px;         /* left + right (or right + left in RTL) */
padding-block-start: 8px;
padding-inline-end: 16px;

/* ─── Border ─── */
border-block: 1px solid gray;
border-inline-start: 3px solid blue; /* left border in LTR, right in RTL */

/* ─── Size ─── */
/* width  → */  inline-size: 300px;
/* height → */  block-size: 200px;
                 min-inline-size: 100px;
                 max-block-size: 600px;

/* ─── Position ─── */
/* top    → */  inset-block-start: 0;
/* bottom → */  inset-block-end: 0;
/* left   → */  inset-inline-start: 0;
/* right  → */  inset-inline-end: 0;
/* inset shorthand: top right bottom left */
                 inset: 0;                /* all four sides */

/* ─── Text alignment ─── */
[dir="rtl"] .card {
  text-align: start;  /* = left in LTR, right in RTL */
  text-align: end;    /* = right in LTR, left in RTL */
}

CSS Subgrid

Subgrid allows nested grid items to participate in their parent grid's track definition. Before subgrid, nested grids created independent track systems — items couldn't align across nested levels. Subgrid solves this for card-based layouts, forms, and complex multi-level designs.

CSS — Subgrid
/* ─── Parent grid ─── */
.card-grid {
  display: grid;
  grid-template-columns: repeat(3, 1fr);
  grid-template-rows: auto 1fr auto;  /* header content footer */
  gap: 20px;
}

/* ─── Card uses subgrid ─── */
.card {
  display: grid;
  /* Inherit parent's ROW tracks — key! */
  grid-row: span 3;       /* span 3 rows */
  grid-template-rows: subgrid; /* use parent's rows! */
  /* Can subgrid columns too: */
  /* grid-template-columns: subgrid; */
}

.card-header { } /* aligns with all other card headers */
.card-body   { } /* aligns and stretches to same height */
.card-footer { } /* always pinned at same baseline */

/* ─── Before subgrid (the problem) ─── */
/* Cards had independent row heights — footers never aligned */

/* ─── With subgrid (the solution) ─── */
/* All card footers align perfectly regardless of content height */

/* ─── Subgrid for form alignment ─── */
.form-grid {
  display: grid;
  grid-template-columns: auto 1fr; /* label | input */
}
.form-group {
  display: grid;
  grid-column: 1 / -1;            /* span all columns */
  grid-template-columns: subgrid; /* inherit label | input columns */
}
CSS3 · Performance

CSS Performance & will-change

CSS performance matters especially for animations. The browser's rendering pipeline has several stages — certain CSS properties are GPU-accelerated and cheap, while others trigger expensive layout recalculations. Understanding this lets you write silky-smooth 60fps animations.

CSS — Performance Optimization
/* ─── The rendering pipeline ─── */
/* JavaScript → Style → Layout → Paint → Composite */

/* ─── COMPOSITE ONLY (GPU, ~0ms) ─── */
/* Only animate these for 60fps! */
.fast {
  transform: translateX(100px);  /* GPU composite ✓ */
  opacity: 0.5;                   /* GPU composite ✓ */
}

/* ─── SKIP LAYOUT (Paint + Composite) ─── */
.medium {
  color: red;          /* triggers repaint, not layout */
  background: blue;    /* triggers repaint, not layout */
  box-shadow: ...;
  border-radius: ...;
}

/* ─── TRIGGERS LAYOUT (avoid animating!) ─── */
.slow {
  /* These force full layout recalc = JANKY */
  width: auto;
  height: 100px;
  top: 0;
  left: 0;
  margin: 10px;
  padding: 20px;
}

/* ─── will-change ─── */
/* Promotes element to its own layer BEFORE animation */
.animated-element {
  will-change: transform;     /* hint to browser */
  will-change: opacity;
  will-change: transform, opacity;

  /* Remove after animation for memory! */
  will-change: auto;  /* reset */
}

/* ─── content-visibility (CSS 2020+) ─── */
.lazy-section {
  /* Skip rendering off-screen sections */
  content-visibility: auto;
  contain-intrinsic-size: 0 800px; /* estimated height */
}

/* ─── contain ─── */
.card {
  contain: layout paint;  /* isolate rendering effects */
  contain: strict;        /* most isolation */
  contain: content;       /* layout+paint+style */
}

/* ─── GPU animation pattern ─── */
.card:hover {
  /* ✗ Avoid: triggers layout */
  /* top: -5px; */

  /* ✓ Use transform instead */
  transform: translateY(-5px) scale(1.02);
  transition: transform 0.2s ease-out;
}

CSS3 · Visual

Custom Scrollbars

CSS provides two approaches to custom scrollbars: the WebKit pseudo-elements (::-webkit-scrollbar) which work in Chrome, Safari, and Edge; and the standards-track scrollbar-width / scrollbar-color from the Scrollbars specification, supported by Firefox and modern browsers.

CSS — Custom Scrollbars
/* ─── Standards (Firefox + modern) ─── */
.element {
  scrollbar-width: thin;   /* auto | thin | none */
  scrollbar-color: #f0a500 #1a1e2a; /* thumb track */
}

/* ─── WebKit (Chrome, Safari, Edge) ─── */
.element::-webkit-scrollbar {
  width: 8px;   /* vertical bar width */
  height: 8px; /* horizontal bar height */
}

.element::-webkit-scrollbar-track {
  background: #1a1e2a;
  border-radius: 4px;
}

.element::-webkit-scrollbar-thumb {
  background: #f0a500;
  border-radius: 4px;
  border: 2px solid #1a1e2a;  /* gap effect */
}

.element::-webkit-scrollbar-thumb:hover {
  background: #ffc107;
}

.element::-webkit-scrollbar-corner {
  background: transparent;
}

/* ─── Overlay scrollbars (modern) ─── */
.overlay {
  overflow: overlay;  /* scroll overlaps content */
}

/* ─── Hide scrollbar but keep scroll ─── */
.hidden-scroll {
  overflow-y: scroll;
  scrollbar-width: none; /* Firefox */
}
.hidden-scroll::-webkit-scrollbar {
  display: none;  /* Chrome/Safari */
}

CSS · Modern

Native CSS Nesting

CSS now supports native nesting (previously only possible with Sass/Less). You can nest selectors directly inside their parent rules, reducing repetition and keeping related styles co-located. Supported in all modern browsers since 2023.

CSS — Native Nesting
/* ─── Basic nesting ─── */
.card {
  background: white;
  border-radius: 8px;
  padding: 20px;

  /* Nested: .card h2 */
  & h2 {
    font-size: 1.5rem;
    color: navy;
  }

  /* Nested: .card p */
  & p { color: #666; }

  /* State: .card:hover */
  &:hover {
    box-shadow: 0 4px 20px rgba(0,0,0,0.1);
    transform: translateY(-2px);
  }

  /* Modifier: .card.featured */
  &.featured {
    border-left: 4px solid gold;
  }

  /* Child combinator: .card > .badge */
  & > .badge {
    position: absolute;
    top: -8px;
  }

  /* Adjacent: .card + .card */
  & + & {
    margin-top: 12px;
  }

  /* Media query inside rule! */
  @media (width >= 768px) {
    padding: 28px;
  }

  /* Deeply nested */
  & .card-footer {
    border-top: 1px solid #eee;

    & button {
      background: gold;
    }
  }
}

/* & = the parent selector (not optional in some cases) */
/* Without &, nested selectors must be compound/complex */
CSS3 · Web Components

Shadow DOM Styling

Shadow DOM creates a scoped DOM tree with encapsulated styles. CSS inside a shadow root doesn't leak out, and external CSS can't easily style shadow elements. Special selectors and CSS custom properties (variables) provide controlled piercing and theming for web components.

CSS — Shadow DOM Selectors
/* ─── Inside shadow root ─── */

/* :host — styles the host element itself */
:host {
  display: block;
  font-family: Georgia, serif;
}

/* :host with condition */
:host(.featured) {
  border: 2px solid gold;
}

/* :host-context — based on ancestor outside shadow */
:host-context([data-theme="dark"]) {
  background: #111;
  color: #fff;
}

/* ::slotted — style slotted (light DOM) content */
::slotted(p) {
  color: blue;
  font-style: italic;
}
::slotted(*) { outline: 2px solid red; }

/* ─── Outside shadow root ─── */

/* CSS variables CROSS the shadow boundary! */
my-component {
  --primary-color: #ff6b35;  /* flows into shadow */
  --font-size: 1.2rem;
}

/* Inside shadow: consume custom properties */
/* :host { color: var(--primary-color, defaultValue); } */

/* CSS Parts API — explicit style hooks */
/* Shadow DOM exposes: */
/* <div part="button"> ... */

/* External can target part: */
my-component::part(button) {
  background: gold;
  padding: 8px 16px;
}
my-component::part(button):hover {
  background: orange;
}

/* @property — typed custom properties */
@property --rotation {
  syntax: '<angle>';
  inherits: false;
  initial-value: 0deg;
}
/* Enables animating custom properties! */
@keyframes spin {
  to { --rotation: 360deg; }
}
CSS3 · Best Practices

The Modern CSS Reset

A CSS reset removes inconsistent browser default styles. The modern approach is minimal and opinionated — not a destructive nuclear reset, but a carefully considered baseline that preserves useful defaults while eliminating the problematic ones.

CSS — Modern Reset (2024)
/* Modern CSS Reset — safe and minimal */

/* ─── Box model ─── */
*, *::before, *::after {
  box-sizing: border-box;
}

/* ─── Reset default margins ─── */
* { margin: 0; }

/* ─── Smooth text rendering ─── */
body {
  -webkit-font-smoothing: antialiased;
  line-height: 1.5;
}

/* ─── Images, video, canvas ─── */
img, picture, video, canvas, svg {
  display: block;
  max-width: 100%;
}

/* ─── Input typography ─── */
input, button, textarea, select {
  font: inherit;
}

/* ─── Text overflow ─── */
p, h1, h2, h3, h4, h5, h6 {
  overflow-wrap: break-word;
}

/* ─── Heading font-size reset ─── */
h1, h2, h3, h4, h5, h6 {
  font-size: inherit;
  font-weight: inherit;
}

/* ─── List style ─── */
ul[class], ol[class] {
  list-style: none;
  padding: 0;
}

/* ─── Stacking context ─── */
#root, #__next {
  isolation: isolate;
}

/* ─── Improve focus visibility ─── */
:focus-visible {
  outline: 3px solid currentColor;
  outline-offset: 3px;
}
:focus:not(:focus-visible) {
  outline: none;
}

/* ─── Reduce motion ─── */
@media (prefers-reduced-motion: reduce) {
  *, *::before, *::after {
    animation-duration: 0.01ms !important;
    animation-iteration-count: 1 !important;
    transition-duration: 0.01ms !important;
    scroll-behavior: auto !important;
  }
}
